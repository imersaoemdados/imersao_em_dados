Pareto Pacote = 
    VAR TOTAL_VENDAS = CALCULATE([Vlr Realizado], ALLSELECTED('fato orcamento_matricial')) 
    VAR TOTAL_ATUAL = [Vlr Realizado]
    VAR tbResumo =  
                SUMMARIZE( 
                    ALLSELECTED('fato orcamento_matricial'), 
                    'dim pacote'[ds_pacote], 
                    "Total", 
                    [Vlr Realizado]
                ) 
    VAR TOTAL_ACUMULADO =  
        SUMX( 
            FILTER(tbResumo, [Total] >= TOTAL_ATUAL), [Total]) 
    RETURN TOTAL_ACUMULADO /TOTAL_VENDAS
